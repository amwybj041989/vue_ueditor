<br>
```javascript 
<Ueditor @ready="editorReady" style="width: 500px;height: 440px;"></Ueditor>
```
<br>

```javascript    
methods: {
    editorReady (instance) {
        instance.setContent('');
        instance.addListener('contentChange', () => {
            this.content = instance.getContent();
        });
      },
    },
```
